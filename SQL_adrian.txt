WHERE  (CLIENT_ID LIKE
                      (SELECT CLIENT_ID
                       FROM      CLIENT AS C
                       WHERE   (CLIENT_NAME LIKE '%' + @val + '%') OR
                                         (CLIENT_SURNAME LIKE '%' + @val + '%') OR
                                         (CLIENT_LASTNAME LIKE '%' + @val + '%') OR
                                         (CLIENT_EGN LIKE '%' + @val + '%') OR
                                         (CLIENT_EMAIL LIKE '%' + @val + '%') OR
                                         (CLIENT_PHONE LIKE '%' + @val + '%')))
GROUP BY DOSSIER_NO, CLIENT_ID, DOSSIER_YEAR, DOSSIER_STATUS, DOSSIER_MIN_BALANCE
